## PlatformKeyMap

Mapping of platform-specific key codes to normalized key actions.

```typescript
interface PlatformKeyMap {
  [keyCode: number]: KeyAction;
}
```

Maps raw numeric key codes to [KeyAction](#keyaction) enums. Each platform has different key code mappings.

### Example

```typescript
const tizenKeyMap: PlatformKeyMap = {
  13: KeyAction.Enter,
  10009: KeyAction.Back,
  37: KeyAction.Left,
  // ...
};
```

## IPlatformAdapter

Platform adapter interface for device abstraction.

```typescript
interface IPlatformAdapter {
  readonly platform: Platform;
  readonly capabilities: PlatformCapabilities;
  readonly deviceInfo: DeviceInfo;
  
  normalizeKeyCode(keyCode: number): KeyAction | null;
  getKeyCodesForAction(action: KeyAction): number[];
  isCodecSupported(codec: string): boolean;
  openExternalLink(url: string): void;
}
```

<ResponseField name="platform" type="Platform">
  Current detected platform.
</ResponseField>

<ResponseField name="capabilities" type="PlatformCapabilities">
  Platform capabilities. See [PlatformCapabilities](#platformcapabilities).
</ResponseField>

<ResponseField name="deviceInfo" type="DeviceInfo">
  Device information. See [DeviceInfo](#deviceinfo).
</ResponseField>

<ResponseField name="normalizeKeyCode" type="(keyCode: number) => KeyAction | null">
  Normalize a raw key code to a KeyAction.
  
  **Parameters:**
  - `keyCode` - Raw keyboard/remote key code from KeyboardEvent
  
  **Returns:**
  - Normalized [KeyAction](#keyaction) or `null` if unmapped
</ResponseField>

<ResponseField name="getKeyCodesForAction" type="(action: KeyAction) => number[]">
  Get platform-specific key codes for a given action.
  
  **Parameters:**
  - `action` - The key action to look up
  
  **Returns:**
  - Array of key codes that map to this action
</ResponseField>

<ResponseField name="isCodecSupported" type="(codec: string) => boolean">
  Check if a specific codec is supported on this platform.
  
  **Parameters:**
  - `codec` - MIME type with codec (e.g., `video/mp4; codecs="hvc1"`)
  
  **Returns:**
  - `true` if codec is supported, `false` otherwise
</ResponseField>

<ResponseField name="openExternalLink" type="(url: string) => void">
  Open an external link in a new tab/window using platform-specific method.
  
  **Parameters:**
  - `url` - The URL to open
</ResponseField>

### Example

```typescript
const adapter: IPlatformAdapter = getPlatformAdapter();

// Normalize key codes
window.addEventListener('keydown', (e) => {
  const action = adapter.normalizeKeyCode(e.keyCode);
  if (action === KeyAction.Back) {
    // Handle back button
  }
});

// Check codec support
if (adapter.isCodecSupported('video/mp4; codecs="hvc1"')) {
  // Use HEVC media file
}

// Open external link
adapter.openExternalLink('https://advertiser.com');
```

## PLATFORM_DETECTION_PATTERNS

Regex patterns for detecting platforms from User-Agent strings.

```typescript
const PLATFORM_DETECTION_PATTERNS: Record<Platform, RegExp[]>;
```

Provides regex patterns to detect each platform from `navigator.userAgent`. Maps [Platform](#platform) enum values to arrays of RegExp patterns.

### Example Structure

```typescript
PLATFORM_DETECTION_PATTERNS[Platform.Tizen] = [
  /Tizen/i,
  /SMART-TV.*Samsung/i
];

PLATFORM_DETECTION_PATTERNS[Platform.WebOS] = [
  /Web0S/i,
  /WebOS/i,
  /LG.*NetCast/i,
  /LGE.*TV/i
];
```

### Detection Patterns by Platform

<ResponseField name="Tizen" type="RegExp[]">
  - `/Tizen/i`
  - `/SMART-TV.*Samsung/i`
</ResponseField>

<ResponseField name="WebOS" type="RegExp[]">
  - `/Web0S/i`
  - `/WebOS/i`
  - `/LG.*NetCast/i`
  - `/LGE.*TV/i`
</ResponseField>

<ResponseField name="Vidaa" type="RegExp[]">
  - `/Vidaa/i`
  - `/VIDAA/i`
  - `/Hisense/i`
</ResponseField>

<ResponseField name="WhaleOS" type="RegExp[]">
  - `/WhaleTV/i`
  - `/Whale/i`
</ResponseField>

<ResponseField name="FireTV" type="RegExp[]">
  - `/AFT/i` - Amazon Fire TV
  - `/AFTS/i` - Fire TV Stick
  - `/AFTM/i` - Fire TV models
  - `/Amazon.*Fire/i`
</ResponseField>

<ResponseField name="Roku" type="RegExp[]">
  - `/Roku/i`

<ResponseField name="Xbox" type="RegExp[]">

<ResponseField name="AndroidTV" type="RegExp[]">
  - `/Android.*TV/i`
  - `/Chromecast/i`
  - `/BRAVIA/i` - Sony TVs
  - `/SHIELD/i` - NVIDIA Shield
</ResponseField>

<ResponseField name="Vizio" type="RegExp[]">
  - `/VIZIO/i`
  - `/SmartCast/i`
</ResponseField>

<ResponseField name="Generic" type="RegExp[]">
  Empty array - fallback for all other platforms
</ResponseField>

### Usage Example

```typescript
function detectPlatform(): Platform {
  const ua = navigator.userAgent;
  
  for (const [platform, patterns] of Object.entries(PLATFORM_DETECTION_PATTERNS)) {
    if (patterns.some(pattern => pattern.test(ua))) {
      return platform as Platform;
    }
  }
  
  return Platform.Generic;
}
```
